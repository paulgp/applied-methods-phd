\begin{frame}{Roadmap for Today}
  \begin{wideitemize}
  \item Today we're focusing on a particular research design: the \emph{leniency design} (aka examiner/judge IV)
  \item These designs harness quasi-random assignment of decision-makers to estimate causal effects of high-stakes treatments
  \item Structure follows Goldsmith-Pinkham, Hull, and Koles\'{a}r (2025), ``Leniency Designs: An Operator's Manual''
  \item Key topics:
    \begin{enumerate}
    \item Setup and intuition
    \item Estimation: why UJIVE?
    \item Heterogeneous effects and monotonicity
    \item Inference
    \item A practical checklist
    \end{enumerate}
  \end{wideitemize}
\end{frame}

\begin{frame}{High-level description of leniency designs}
    \begin{columns}[onlytextwidth, T] % align columns
      \begin{column}{.6\textwidth}
        \begin{wideitemize}
        \item In many applications, there is an administrator, judge,
          or examiner who plays an important role in deciding an
          outcome
        \item These outcomes include:
          \begin{itemize}
          \item bail
          \item bankruptcy
          \item getting a loan
          \item parole
          \item disability insurance
          \item patent granting
          \item cancer screening
          \end{itemize}
        \item In many cases, this examiner is effectively randomly
          assigned, \emph{and} there is wide-range of differences (and
          discretion) in how likely they are to decide the outcome
        \end{wideitemize}
      \end{column}%
      \hfill%
      \begin{column}{.4\textwidth}
        \only<2->{
        \begin{wideitemize}
        \item Key ingredients:
          \begin{enumerate}
          \item as-good-as-random assignment of decision-maker
          \item discretion over a (typically binary) treatment
          \item heterogeneity in decision-maker behavior (``leniency'')
          \end{enumerate}
        \end{wideitemize}
        }
      \end{column}%
    \end{columns}
\end{frame}

  \begin{columns}[onlytextwidth, T] % align columns

        \item The particular magistrate that a defendant faces

        \item Hence, the randomness occurs from \emph{the time of

\begin{frame}{A simple motivating example: Two examiners}
  \begin{wideitemize}
  \item Consider a stylized patent setting (Farre-Mensa et al.\ (2020)):
    \begin{itemize}
    \item $n$ patent applications, randomly assigned to one of two examiners $s$ (``soft'') or $t$ (``tough'')
    \item Treatment $x_{i} \in \{0,1\}$: patent granted or not
    \item Outcome $y_{i}$: future patent applications by the firm
    \end{itemize}
  \item Outcome equation (constant effects for now):
    \begin{equation*}
      y_{i} = \gamma + \beta x_{i} + \varepsilon_{i}
    \end{equation*}
  \item OLS is biased: firms with high $\varepsilon_{i}$ (better scientists, more R\&D) submit stronger applications $\Rightarrow$ positive selection bias
  \item Leniency design idea: use the randomly-assigned examiner's tendency to grant patents (their \emph{leniency}) as an instrument for $x_{i}$
  \item With two examiners: using the examiner dummy $z_{i}$ as an instrument is numerically identical to using estimated leniency $\hat{\ell}_{i}$
  \end{wideitemize}
\end{frame}

\begin{frame}{Notation: Many decision-makers and controls}
  \begin{wideitemize}
  \item Now consider $K$ examiners. Assignment captured by instrument vector $z_{i}$ (examiner dummies)
  \item \textbf{Necessary controls} $w_{i}$: variables needed to make assignment as-good-as-random
    \begin{itemize}
    \item E.g.\ art unit $\times$ year FE (patents), court $\times$ time FE (judges)
    \end{itemize}
  \item First-stage equation:
    \begin{equation*}
      x_{i} = z_{i}'\pi + w_{i}'\delta + \nu_{i}
    \end{equation*}
  \item Population first-stage fitted values give \textbf{absolute leniency}: $\ell_{i} = z_{i}'\pi + w_{i}'\delta$
  \item After residualizing for $w_{i}$, we get \textbf{relative leniency}: $\tilde{\ell}_{i} = \tilde{z}_{i}'\pi$
    \begin{itemize}
    \item Measures leniency of assigned examiner \emph{relative to} other examiners who could have been assigned
    \item E.g.\ examiner's granting rate minus art unit average granting rate
    \end{itemize}
  \end{wideitemize}
\end{frame}

\begin{frame}{What's the instrument that people have used?}
    \begin{columns}[onlytextwidth, T] % align columns
      \begin{column}{.5\textwidth}
        \begin{wideitemize}
        \item In many papers, the measure that has been used is a
          ``leniency'' metric for judges: e.g.\ the $\hat{\ell}_{i}$
        \item This is the overall average leniency across judges in
          DGPY [note, not observation weighted -- hence outliers]
        \item What's the problem with just the overall?
          \begin{itemize}
          \item Judges are not random overall -- location (and time)
            specific
          \end{itemize}
        \item Must control for $w_{i}$ to get \emph{relative} leniency
        \end{wideitemize}
      \end{column}%
      \hfill%
      \begin{column}{.5\textwidth}
        \includegraphics[width=\linewidth]{images/leniency_overall.png}
      \end{column}%
    \end{columns}
\end{frame}

\begin{frame}{Relative leniency after residualizing}
    \begin{columns}[onlytextwidth, T] % align columns
      \begin{column}{.5\textwidth}
        \begin{wideitemize}
        \item Once you residualize for controls, you get a
          much more recentered object
        \item This captures within-location variation and gives true examiner-specific variation
        \item Due to the location effect, we
          can't estimate ``true'' judge leniency -- just relative
          leniency within an office
        \end{wideitemize}
      \end{column}%
      \hfill%
      \begin{column}{.5\textwidth}
        \includegraphics[width=\linewidth]{images/leniency_res.png}
      \end{column}%
    \end{columns}
\end{frame}

\begin{frame}{The infeasible ideal and the bias problem}
  \begin{wideitemize}
  \item If we knew $\pi$, we could use relative leniency $\tilde{\ell}_{i} = \tilde{z}_{i}'\pi$ as the instrument:
    \begin{equation*}
      \hat{\beta}^{*} = \frac{\sum_{i} y_{i} \tilde{\ell}_{i}}{\sum_{i} x_{i} \tilde{\ell}_{i}}
    \end{equation*}
  \item Since $\tilde{\ell}_{i}$ is as-good-as-random and uncorrelated with $\varepsilon_{i}$, this is approximately unbiased
  \item But we don't know $\pi$! Using sample estimates $\hat{\pi}$ gives us 2SLS:
    \begin{itemize}
    \item The estimated leniency $\hat{\ell}_{i} = \tilde{z}_{i}'\hat{\pi}$ is constructed \emph{in part from observation $i$'s own treatment}
    \item This creates a mechanical correlation with $\varepsilon_{i}$ $\Rightarrow$ many-IV bias toward OLS
    \end{itemize}
  \end{wideitemize}
\end{frame}


\begin{frame}{How severe is the many-IV bias?}
  \begin{wideitemize}
  \item Under homoskedasticity, 2SLS bias $\approx$ OLS bias $\times \frac{1}{(1-R^{2}) \times E[F]}$
    \begin{itemize}
    \item $E[F]$: expected first-stage $F$ statistic; $R^{2}$: partial R-squared from instruments
    \item Since $R^{2} \approx 0$ typically, the magnitude of $E[F]$ tells us how much smaller 2SLS bias is vs.\ OLS
    \end{itemize}
  \item This justifies the $F > 10$ rule of thumb (Staiger and Stock, 1997): if $E[F] < 10$, 2SLS bias exceeds 10\% of OLS bias
  \item With many examiners, $F$ can be modest even when leniency explains meaningful variation, because the formula divides by $K$
  \end{wideitemize}
  % \only<1>{\begin{center}\includegraphics[width=0.55\linewidth]{images/manyiv_hull1.png}\end{center}}
  % \only<2>{\begin{center}\includegraphics[width=0.55\linewidth]{images/manyiv_hull2.png}\end{center}}
  % \only<3>{\begin{center}\includegraphics[width=0.55\linewidth]{images/manyiv_hull3.png}\end{center}}
\end{frame}

\begin{frame}{The leave-out solution and why manual leniency is fragile}
  \begin{wideitemize}
  \item The bias comes from using own treatment to estimate leniency
  \item Common practice: use a ``leave-one-out'' (LOO) leniency measure $\hat{\ell}_{-i}$
    \begin{itemize}
    \item Excludes observation $i$'s own treatment from the examiner average
    \item Removes the mechanical correlation with $\varepsilon_{i}$
    \end{itemize}
  \item But this ``manual'' leniency construction is fragile!
    \begin{itemize}
    \item LOO of \emph{absolute} leniency $\Rightarrow$ residualizing for $w_{i}$ reintroduces own-observation bias through the control means
    \item This is JIVE (Phillips and Hale, 1977); its bias can be comparable to 2SLS when there are many controls:
      \begin{equation*}
        \text{JIVE bias} \approx \text{2SLS bias} \times \frac{-E[F] \times L}{(E[F]-1)K - L}
      \end{equation*}
      where $L$ = number of controls, $K$ = number of instruments
    \end{itemize}
  \item \textbf{Takeaway}: subtle variations in how the leniency measure is constructed and accounts for covariates can have large consequences
  \end{wideitemize}
\end{frame}

\begin{frame}{Use UJIVE!}
  \begin{wideitemize}
  \item \textbf{UJIVE} (Unbiased Jackknife IV, Koles\'{a}r 2013): the purpose-built solution
    \begin{itemize}
    \item Estimates \emph{relative} leniency using leave-out: $\hat{\ell}_{-i} = \tilde{z}_{i}'\hat{\pi}_{-i}$
    \item First residualize instruments for $w_{i}$, \emph{then} do leave-out estimation
    \item Avoids the own-observation bias of both 2SLS and JIVE
    \end{itemize}
  \item Approximately unbiased even with many examiners and many controls
  \item Remains consistent even with weak instruments (as long as $\sqrt{K} \times E[F]$ is large)
  \item No need to report first-stage $F$ as a diagnostic: UJIVE doesn't suffer from the bias that $F > 10$ screens for
  \item Don't construct leniency manually -- use UJIVE as a one-step estimator
    \begin{itemize}
    \item R package: \texttt{ManyIV} at \texttt{github.com/kolesarm/ManyIV}
    \end{itemize}
  \end{wideitemize}
\end{frame}

\begin{frame}{Comparison of estimators}
  \begin{wideitemize}
  \item \textbf{2SLS with examiner dummies}: biased toward OLS with many examiners
  \item \textbf{JIVE} (leave-out absolute leniency): bias reintroduced by many controls
  \item \textbf{IJIVE} (Ackerberg and Devereux, 2009): reverses order---residualize \emph{then} leave-out; similar to UJIVE in practice
  \item \textbf{FEJIV} (Chao, Swanson, Woutersen, 2023): minimal MSE leniency with orthogonality to controls; attractive but computationally demanding 
  \item \textbf{UJIVE}: recommended default
    \begin{itemize}
    \item Minimal MSE without orthogonality constraint
    \item Simple, established unbiasedness properties
    \item Works with many instruments \emph{and} many controls
    \end{itemize}
  \item \textbf{Manual leniency IV}: widely advised against (like ``manual 2SLS'')
  \end{wideitemize}
\end{frame}

\begin{transitionframe}
  \begin{center}
    {\Huge \textcolor{blue}{Part 3: Heterogeneous Effects and Monotonicity}}
  \end{center}
\end{transitionframe}


\begin{frame}{What does UJIVE estimate with heterogeneous effects?}
  \begin{wideitemize}
  \item So far we assumed constant $\beta$. In practice, treatment effects $\beta_{i}$ vary across individuals
  \item The LATE theorem (Imbens and Angrist, 1994) provides a causal interpretation under:
    \begin{enumerate}
    \item As-good-as-random assignment (conditional on controls)
    \item Exclusion restriction
    \item \textbf{Monotonicity}: if examiner $a$ would grant in a case where $b$ wouldn't, then $a$ is more lenient in \emph{all} cases
    \end{enumerate}
  \item With two examiners: divides population into \textbf{compliers} (treatment depends on assignment) and \textbf{non-responders} (always/never treated)
  \item Monotonicity rules out \textbf{defiers} (treated only by the tough examiner)
  \item IV estimates the average treatment effect for compliers (LATE)
  \end{wideitemize}
\end{frame}

\begin{frame}{Monotonicity with many examiners}
  \begin{columns}[onlytextwidth, T] % align columns
    \begin{column}{.55\textwidth}
      \begin{wideitemize}
      \item With many examiners, standard monotonicity is very strong:
        requires all examiners to rank \emph{every} case identically
      \item Equivalent to assuming examiners differ only in their \emph{cutoff}, not their ranking criteria
      \item Two core concerns:
        \begin{enumerate}
        \item Multi-dimensional evaluation criteria (e.g.\ novelty vs.\ usefulness vs.\ non-obviousness)
        \item Skill variation induces ranking differences through mistakes
        \end{enumerate}
      \end{wideitemize}
    \end{column}%
    \hfill%
    \begin{column}{.45\textwidth}
      \includegraphics[width=\linewidth]{images/imbensangrist_judge.png}
    \end{column}%
  \end{columns}
\end{frame}

\begin{frame}{Average monotonicity: a weaker condition}
  \begin{wideitemize}
  \item With many examiners, UJIVE estimates a weighted average of pairwise-complier treatment effects, weighted by squared leniency differences
    \begin{itemize}
    \item Larger complier groups get more weight
    \end{itemize}
  \item Without monotonicity, defier groups get \emph{negative} weight $\Rightarrow$ problematic
  \item But each individual appears in many pairwise comparisons. Total weight on individual $i$ is their \emph{average} complier/defier status
  \item \textbf{Average monotonicity} (Frandsen et al.\ 2023): the average leniency of examiners who would grant exceeds the average leniency of those who would deny
    \begin{itemize}
    \item Weaker than standard monotonicity: allows some pairwise violations
    \item Sufficient for positive individual weights in UJIVE
    \item With two examiners, reduces to standard monotonicity
    \end{itemize}
  \end{wideitemize}
\end{frame}

\begin{frame}{What if monotonicity is violated?}
  \begin{columns}[onlytextwidth, T] % align columns
    \begin{column}{.5\textwidth}
      \begin{wideitemize}
      \item Even with some negative weights, it only matters if negatively-weighted observations have systematically different treatment effects
      \item Sigstad (2025): in judicial panels where multiple judges see the same case, monotonicity is technically violated, but ranking disagreements are not severe enough to create substantial bias
      \item Key question: are the violations \emph{large enough} to matter?
      \end{wideitemize}
    \end{column}%
    \hfill%
    \begin{column}{.5\textwidth}
      \includegraphics[width=\linewidth]{images/gphk_avgmonotonicity.png}
    \end{column}%
  \end{columns}
\end{frame}

\begin{frame}{Testing average monotonicity with UJIVE}
  \begin{wideitemize}
  \item Key insight (building on Abadie 2002, Kitagawa 2015): UJIVE can \textbf{characterize compliers}
  \item For a binary variable $v_{i}$ (determined before assignment), construct $\tilde{y}_{i} = v_{i} \times x_{i}$ and run UJIVE with $\tilde{y}_{i}$ as outcome
    \begin{itemize}
    \item Under average monotonicity, UJIVE estimates a convex-weighted average of $v_{i}$ for compliers
    \item This estimate must lie in $[0,1]$ since $v_{i} \in \{0,1\}$
    \item A negative estimate or one exceeding 1 $\Rightarrow$ evidence against average monotonicity
    \end{itemize}
  \item Can also use the original outcome: set $\tilde{y}_{i} = y_{i} \times x_{i}$ to estimate weighted average of \emph{treated} complier outcomes
  \item Create indicators for each outcome value (e.g.\ ``zero future applications'') and check all UJIVE estimates are positive
  \item This test inherits UJIVE's unbiasedness with many examiners/controls
  \end{wideitemize}
\end{frame}


\begin{frame}{Kitagawa (2015) result (background)}
  \begin{columns}[onlytextwidth, T] % align columns
    \begin{column}{.5\textwidth}
      \begin{itemize}
      \item Consider binary endogenous treatment $D$ and a discrete instrument $Z$.
      \item Joint \emph{testable} assumption: instrument is valid and monotonicity. Why?
        \begin{itemize}
        \item Consider $p(y,d) = Pr(Y = y, D = d | Z = 1)$, $q(y,d) = Pr(Y = y, D = d | Z = 0)$
        \item Let $P$ and $B$ be the probability over sets
        \item Imbens and Rubin (1997) show
          \vspace{-10pt}
          \begin{align*}
            P(B, 1) - Q(B,1) &= Pr(Y_{1} \in B, D_{1} > D_{0})\\
            P(B, 0) - Q(B,0) &= Pr(Y_{0} \in B, D_{1} > D_{0})\\                        
          \end{align*}
        \end{itemize}
        \vspace{-10pt}                
      \item Testable implication:
          \vspace{-10pt}        
          \begin{align*}
            P(B, 1) - Q(B,1) &\geq 0\\
            P(B, 0) - Q(B,0) &\geq 0
          \end{align*}
      \end{itemize}
    \end{column}%
    \hfill%
    \begin{column}{.5\textwidth}
            \only<1>{\includegraphics[width=\linewidth]{images/kitagawa_valid.png}}
            \only<2>{      \includegraphics[width=\linewidth]{images/kitagawa_invalid.png}}
    \end{column}%
  \end{columns}
\end{frame}

    {\Huge \textcolor{blue}{Part 4: Inference}}
  \end{center}
\end{transitionframe}

\begin{frame}{Standard errors: cluster or not?}
  \begin{wideitemize}
  \item The assignment mechanism should guide the choice of standard errors
  \item \textbf{Individual-level randomization} (e.g.\ patent examiners assigned idiosyncratically within art unit-year):
    \begin{itemize}
    \item Heteroskedasticity-robust (HC) standard errors are appropriate
    \item No need to cluster by examiner
    \end{itemize}
  \item \textbf{Group-level assignment} (e.g.\ all cases in a month assigned to the same examiner):
    \begin{itemize}
    \item Cluster at the group level (e.g.\ month)
    \item May also need leave-cluster-out modification of UJIVE (Frandsen et al.\ 2025)
    \end{itemize}
  \item Rule of thumb: clustering is needed when groups of observations are assigned together
  \item UJIVE computes correct heteroskedastic-robust SEs automatically
  \item Clustering justification also has implications for the \emph{estimator}, not just inference
  \end{wideitemize}
\end{frame}

\begin{frame}{Don't be fooled by small 2SLS standard errors}
  \begin{wideitemize}
  \item With many examiners, 2SLS standard errors can be \emph{much} smaller than UJIVE SEs
  \item This reflects statistical pathology, not efficiency:
    \begin{itemize}
    \item 2SLS estimates are pulled toward OLS (biased)
    \item 2SLS standard errors are also pulled toward OLS SEs (too small)
    \item Creates a false sense of precision around biased estimates
    \end{itemize}
  \item Even leave-out leniency measures (manual construction) can produce SEs that don't reflect estimation error in the leniency measure
  \item UJIVE delivers both unbiased point estimates \emph{and} standard errors that accurately capture sampling variation
  \end{wideitemize}
\end{frame}

\begin{transitionframe}
  \begin{center}
    {\Huge \textcolor{blue}{Part 5: A Practical Checklist}}
  \end{center}
\end{transitionframe}

\begin{frame}{The 5-step checklist (GHK 2025)}
  \begin{enumerate}
  \addtolength{\itemsep}{8pt}
  \item \textbf{Identify necessary controls} for as-good-as-random assignment, and determine appropriate estimator and standard errors
  \item \textbf{Verify balance} on predetermined observables
  \item \textbf{Estimate treatment effects} by UJIVE (and compare to 2SLS/OLS)
  \item \textbf{Test average monotonicity} to assess LATE interpretation
  \item \textbf{Characterize compliers} to assess external validity
  \end{enumerate}
\end{frame}

\begin{frame}{Step 1: Identify necessary controls and design}
  \begin{wideitemize}
  \item Begin with institutional knowledge that justifies quasi-random assignment
    \begin{itemize}
    \item What is the assignment mechanism? What must be conditioned on?
    \item Patent example: assignment within art units is effectively random, conditional on the set of examiners $\Rightarrow$ art unit $\times$ year FE
    \item Bail example: rotating schedule $\Rightarrow$ time-of-day variation
    \end{itemize}
  \item These are \emph{necessary controls}---must be included in every specification
  \item Separately, may include \emph{precision controls} (pre-assignment covariates) to reduce SEs
  \item Also argue for the exclusion restriction from institutional knowledge
    \begin{itemize}
    \item Patent examiners make narrow approval decisions $\Rightarrow$ unlikely to directly affect future innovation
    \end{itemize}
  \item Assignment mechanism determines clustering level for SEs (and potentially for leave-cluster-out UJIVE)
  \end{wideitemize}
\end{frame}

\begin{frame}{Step 2: Verify balance on observables}
  \begin{columns}[onlytextwidth, T] % align columns
    \begin{column}{.5\textwidth}
      \begin{wideitemize}
      \item As-good-as-random assignment implies average predetermined characteristics are equal across examiners (conditional on controls)
      \item \textbf{Best practice}: run UJIVE specification with each predetermined covariate as the outcome
        \begin{itemize}
        \item Same instruments, same controls as main specification
        \item Avoids finite-sample biases from regressing on constructed leniency
        \end{itemize}
      \item Magnitude of imbalance directly translates to potential bias in treatment effects
      \end{wideitemize}
    \end{column}%
    \hfill%
    \begin{column}{.5\textwidth}
      \includegraphics[width=\linewidth]{images/gphk_balance_table.png}
    \end{column}%
  \end{columns}
\end{frame}

\begin{frame}{Step 3: Estimate treatment effects}
  \begin{wideitemize}
  \item Use UJIVE as the preferred estimator
  \item Report alongside alternative estimators for comparison:
    \begin{itemize}
    \item OLS (benchmark for selection bias)
    \item 2SLS with examiner dummies (shows many-IV bias)
    \item Manual leniency IV if used in prior literature (shows construction sensitivity)
    \end{itemize}
  \item Pattern to look for:
    \begin{itemize}
    \item 2SLS estimates between OLS and UJIVE $\Rightarrow$ 2SLS pulled toward OLS as predicted
    \item 2SLS SEs much smaller than UJIVE SEs $\Rightarrow$ false precision from bias
    \end{itemize}
  \item In Farre-Mensa et al.\ reanalysis: UJIVE finds significant positive effects of patent approval on future innovation, but 2SLS SEs are 3--4x smaller (pathological)
  \end{wideitemize}
\end{frame}

\begin{frame}{Step 4: Test average monotonicity}
  \begin{wideitemize}
  \item For each possible outcome value, create indicator and interact with treatment:
    \begin{itemize}
    \item E.g.\ $\tilde{y}_{i} = \mathbf{1}(y_{i} = k) \times x_{i}$ for each $k$
    \item Run UJIVE with same instruments and controls
    \item All estimates should be positive (and between 0 and 1)
    \end{itemize}
  \item This tests the weaker \emph{average} monotonicity, not full IA94 monotonicity
  \item Uses the same UJIVE framework---inherits robustness to many instruments/controls
  \item Frandsen et al.\ (2023) test of strict monotonicity also available (\texttt{testjfe} in Stata), but:
    \begin{itemize}
    \item Rules out many decision-makers
    \item Requires bounded outcomes
    \item Tests the stronger condition
    \end{itemize}
  \end{wideitemize}
\end{frame}

\begin{frame}
  \includegraphics[width=\linewidth]{images/monotonicity_ujive.png}
\end{frame}

\begin{frame}{Step 5: Characterize compliers for external validity}
  \begin{wideitemize}
  \item Even with a valid LATE, we want to know: are compliers representative of the broader population?
  \item Use UJIVE with modified outcomes to characterize compliers:
    \begin{itemize}
    \item For each baseline covariate $v_{i}$, run UJIVE with $\tilde{y}_{i} = v_{i} \times x_{i}$ as outcome
    \item Recovers weighted average of $v_{i}$ for treated compliers
    \item For untreated compliers: use $v_{i} \times (1 - x_{i})$ instead
    \item Pool efficiently: use $\tilde{x}_{i} = 2x_{i} - 1$ and regress $v_{i} \times \tilde{x}_{i}$ on $\tilde{x}_{i}$ (Angrist et al.\ 2023)
    \end{itemize}
  \item Compare complier means to population means
    \begin{itemize}
    \item If similar: LATE likely approximates ATE
    \item If different: effects may not generalize
    \end{itemize}
  \item Complier estimates within $[0,1]$ for binary $v_{i}$ also supports average monotonicity
  \end{wideitemize}
\end{frame}

%% CONCLUSION

\begin{frame}{Summary: Key recommendations}
  \begin{wideitemize}
  \item \textbf{Estimation}: Use UJIVE, not manual leniency or 2SLS
    \begin{itemize}
    \item Avoids many-IV bias and many-control bias simultaneously
    \item R package: \texttt{ManyIV}
    \end{itemize}
  \item \textbf{Controls}: Identify necessary controls from institutional knowledge; precision controls optional
  \item \textbf{Inference}: Heteroskedastic-robust SEs when individual-level assignment; cluster when group assignment
  \item \textbf{Monotonicity}: Test average monotonicity (weaker than standard) via UJIVE
  \item \textbf{External validity}: Characterize compliers using UJIVE with modified outcomes
  \item \textbf{Don't} construct leniency measures manually -- one-step UJIVE is simpler and safer
  \end{wideitemize}
\end{frame}
